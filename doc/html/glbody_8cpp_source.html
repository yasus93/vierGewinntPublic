148 bool GLBody::readBinaryModelFile(const QString &dataFileName, QVector<GLPoint> * pointContainer, QVector<IndexType> * indexContainer)
149 {
150  makeSurface(pointContainer, indexContainer);
151  QFile binaryFile( dataFileName );
152  if( !binaryFile.open( QIODevice::ReadOnly )) {
153  qDebug() << m_name + " GLBody::readBinaryFile: could not open file: " + dataFileName;
154  return false;
155  }
156  // Die skip Methode braucht man hier nicht, da jedesmal wenn was gelesen wurde bleibt die Position bestehen
157  QDataStream dataStream( & binaryFile );
158  unsigned int formatVersion = 0;
159  int numberOfPoints = 0;
160  int numberOfIndices = 0;
161 
162  dataStream.readRawData(reinterpret_cast<char*>( & formatVersion), 4);
163  if(formatVersion != 10 && formatVersion != 11)
164  {
165  qDebug() << m_name << " GLBody::readBinaryModelFile: Error! Format version mismatch. Version in File: " << formatVersion;
166  return false;
167  }
168  dataStream.readRawData(reinterpret_cast<char*>( & numberOfPoints), 4);
169  int pointBytes = numberOfPoints * static_cast<int>(sizeof(GLPoint));
170  dataStream.readRawData(reinterpret_cast<char*>( & numberOfIndices), 4);
171  int indexBytes = 0; //numberOfIndices * sizeof(GLshort );
172 
173  m_firstPoint = pointsSize();
174  m_points->resize(m_points->size() + numberOfPoints);
175  dataStream.readRawData(reinterpret_cast<char*>((*m_points)[m_firstPoint].vertexPointer()), pointBytes);
176  m_nextPoint = pointsSize();
177 
178  m_firstIndex = indicesSize();
179  if(formatVersion == 10){
180  indexBytes = numberOfIndices * static_cast<int>(sizeof(GLuint ));
181  QVector<IndexType> indices;
182  indices.resize(numberOfIndices);
183  dataStream.readRawData(reinterpret_cast<char*>( &(indices[0]) ), indexBytes);
184  for(int i = 0; i < numberOfIndices; i++) //convert to short and append
185  m_indices->append(indices[i]);
186  }
187  else if(formatVersion == 11){
188  indexBytes = numberOfIndices * static_cast<int>(sizeof(IndexType ));
189  m_indices->resize(m_indices->size() + numberOfIndices);
190  dataStream.readRawData(reinterpret_cast<char*>( &(*m_indices)[m_firstIndex]), indexBytes);
191  }
192  m_nextIndex = indicesSize();
193  //now reposition index numbers
194  if(m_firstPoint != 0)
195  {
196  for(int i = m_firstIndex; i < m_nextIndex; i++)
197  m_indices->data()[i] = m_indices->at(i) + m_firstPoint;
198  }
199  //move model zero to coordinate origin
200  if(m_modelOffset != v_Zero)
201  move(-m_modelOffset); // will call findMinMaxCoordinates() too
202  else findMinMaxCoordinates();
203  return true;
204 }

206 void GLBody::rotateModelPoints(const QVector3D &axisCenter, const QVector3D &axisDirection, float angle)
207 {
208  if(!m_surfaceIsValid)
209  return;
210  QMatrix4x4 rotator;
211  rotator.rotate(angle, axisDirection);
212  rotator.translate(-axisCenter);
213 
214  QMatrix4x4 translator;
215  translator.translate(axisCenter);
216  QMatrix4x4 transformer = translator * rotator;
217  QMatrix4x4 normalMatrix = QMatrix4x4 (transformer.normalMatrix());
218  for(int i = m_firstPoint; i < m_nextPoint; i++)
219  (*m_points)[i].transform(transformer, normalMatrix);
220  findMinMaxCoordinates();
221 }

231 void GLBody::draw(GLESRenderer * renderer, bool useBuffers){
232  if(!m_surfaceIsValid)
233  makeSurface(nullptr, nullptr);
234  if(!m_points || m_points->size() == 0) //check for existing points
235  {
236  qDebug()<< m_name + " GLBody::draw ERROR: No points!";
237  return;
238  }
239  bool textureEnabled = renderer->isTextureEnabled();
240  bool lightingEnabled = renderer->isLightingEnabled();
241  bool colorArrayEnabled = renderer->isColorArrayEnabled();
242 
243  //set colors
244  renderer->setAmbientAndDiffuseColor(m_color);
245  //set transformations
246  calculateDrawMatrix();
247  renderer->pushMvMatrix();
248  renderer->addTransformation(m_drawMatrix);

252  glFuncs.glActiveTexture(GL_TEXTURE0); //use texture unit 0
253  glBindTexture(GL_TEXTURE_2D, m_textureIds[m_activeTextureIndex]);
254  renderer->setTextureEnabled(true);
255  }
256  else renderer->setTextureEnabled(false);
257 
258  void * indexAddress = nullptr;
259  if(useBuffers){
260  indexAddress = reinterpret_cast<void *>( m_firstIndex * sizeof(IndexType));
261  activateAttributeBuffers(renderer);
262  }
263  else{
264  activateAttributeArrays(renderer);
265  indexAddress = &m_indices->data()[m_firstIndex];
266  }
267 
268  if(m_nextIndex - m_firstIndex > 0) //we have indexes defined
269  glDrawElements(m_drawingMode, m_nextIndex - m_firstIndex,
270  GL_UNSIGNED_SHORT, indexAddress);
271  else
272  glDrawArrays(m_drawingMode, m_firstPoint, m_nextPoint - m_firstPoint);
273 
274  if(lightingEnabled)
275  renderer->setLightingEnabled(false);
276  if(!colorArrayEnabled)
277  renderer->setColorArrayEnabled(true);
278  renderer->setTextureEnabled(false);
279  if(m_showNormals){
280  if(useBuffers)
281  indexAddress = reinterpret_cast<void *>( m_firstNormalIndex * sizeof(IndexType));
282  else indexAddress = &m_indices->data()[m_firstNormalIndex];
283  glDrawElements(GL_LINES, m_nextNormalIndex -m_firstNormalIndex,
284  GL_UNSIGNED_SHORT, indexAddress);
285  }
286 
287  if(m_showFrame)
288  {
289  //draw the frame
290  GLfloat lineWidth[4]; //4 floats, just for safety
291  glGetFloatv(GL_LINE_WIDTH, lineWidth);
292  glLineWidth(3.0);
293  if(useBuffers)
294  indexAddress = reinterpret_cast<void *>( m_firstFrameIndex * sizeof(IndexType));
295  else indexAddress = &m_indices->data()[m_firstFrameIndex];
296  glDrawElements(GL_LINES, m_nextFrameIndex -m_firstFrameIndex,
297  GL_UNSIGNED_SHORT, indexAddress);
298  //Draw the hit line
299  if(m_firstHitPoint != 0)
300  {
301  glLineWidth(5.0);
302  glDrawArrays(GL_LINES, m_firstHitPoint, m_nextHitPoint - m_firstHitPoint);
303  }
304  glLineWidth(lineWidth[0]);
305  }
306 
307  if(isSelected())
308  {
309  GLColorRgba pointColor(m_color.inverted());
310  renderer->setAmbientAndDiffuseColor(pointColor);
311 #ifdef GLES
312  glDrawArrays(GL_POINTS, m_firstPoint, m_nextPoint - m_firstPoint);
313 #else
314  float oldPointSize = renderer->pointSize();
315  renderer->setPointSize(5);
316  glDrawArrays(GL_POINTS, m_firstPoint, m_nextPoint - m_firstPoint);
317  renderer->setPointSize(oldPointSize);
318 #endif
319  }
320 
321  //cleanup
322  renderer->setLightingEnabled(lightingEnabled);
323  renderer->setColorArrayEnabled(colorArrayEnabled);
324  renderer->setTextureEnabled(textureEnabled);
325  renderer->disableAttributeArrays();
326  renderer->popMvMatrix();
327 }

581 bool GLBody::calculateIntersection(const QVector3D &p1, const QVector3D &p2, const QVector3D &p3, const QVector3D &pLine, const QVector3D &lineDirection, float *a, float *b, QVector3D *intersection)
582 {
583  double denominator = 0.0;
584  //We solve the linear equation a*p12 + b * p13 + lambda * lineDirection = pLine - P1
585  //We only need a and b.
586  //An intersection exists if denominator determinant is != 0.
587  //The intersection is inside the triangle if a > 0 and b > 0 and a+b < 1
588  QVector3D p12 = p2 - p1;
589  QVector3D p13 = p3 - p1;
590  QVector3D rightSide = pLine - p1;
591  QMatrix4x4 m; //Determinant function is available for 4x4 matrix only.
592  QMatrix4x4 denominatorMatrix;
593  denominatorMatrix.setColumn(0, p12);
594  denominatorMatrix.setColumn(1, p13);
595  denominatorMatrix.setColumn(2, lineDirection);
596  denominator = denominatorMatrix.determinant();
597  if(fabs(denominator) < 0.00001) //too small for an intersection
598  return false;
599  m = denominatorMatrix;
600  m.setColumn(0, rightSide); //replace the a-Column with right side of equation
601  *a = static_cast<float>(m.determinant() / denominator);
602  m = denominatorMatrix;
603  m.setColumn(1,rightSide); //replace the b-Column with right side of equation
604  * b = static_cast<float>( m.determinant() / denominator);
605  if(intersection != nullptr)
606  * intersection = p1 + (*a) * p12 + (*b) * p13;
607  return true;
608 }
