370  const QVector3D & normal, const QVector3D & texCoord,
371  const GLColorRgba& color)
372 {
373  int ticks = static_cast<int>(floor(static_cast<double>(length)));
374  float tickLength = 0.2f;
375  m_points.append(GLPoint(origin - axis * length, normal, texCoord,color));
376  m_points.append(GLPoint(origin + axis * length, normal, texCoord,color));

378  {
379  m_points.append(GLPoint(origin + axis * tick, normal, texCoord, color));
380  if(tick%5 == 0)
381  m_points.append(GLPoint(origin + axis * tick + normal * tickLength * 2, normal, texCoord, color));
382  else m_points.append(GLPoint(origin + axis * tick + normal * tickLength, normal, texCoord, color));
383  }
384  QVector3D normal2 = QVector3D::crossProduct(axis, normal).normalized();
385  for(int tick = -ticks; tick <= ticks; tick++)
386  {
387  m_points.append(GLPoint(origin + axis * tick, normal2, texCoord, color));
388  if(tick%5 == 0)
389  m_points.append(GLPoint(origin + axis * tick + normal2 * tickLength * 2, normal2, texCoord, color));
390  else m_points.append(GLPoint(origin + axis * tick + normal2 * tickLength, normal2, texCoord, color));
