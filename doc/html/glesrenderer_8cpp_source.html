217 QVector3D GLESRenderer::unProjectViewportPoint(const QVector3D & vWin)
218 {
219  QVector3D vClip = viewportToClip(vWin);
220  if(!m_invertedMvpMatrixValid)
221  {
222  m_invertedMvpMatrix = m_mvpMatrix.inverted();
223  m_invertedMvpMatrixValid = true;
224  }
225  ShaderDebugger::debugMatrix4x4(m_mvpMatrix, "MVP Matrix:");
226  ShaderDebugger::debugMatrix4x4(m_invertedMvpMatrix, "Inverted MVP Matrix:");
227  QVector3D result = m_invertedMvpMatrix * vClip;
228  ShaderDebugger::debugVector3D(result, "Vector in model space:");
229  return result;
230 }

235 QVector3D GLESRenderer::viewportToClip(const QVector3D & vWin)
236 {
237  //reverse viewport transformation, for original code see below
238  float xClip = (vWin.x() - static_cast<float>(m_viewportX)) /static_cast<float>(m_viewportW) * 2.0f - 1.0f;
239  float yClip = (vWin.y() - static_cast<float>(m_viewportY)) /static_cast<float>(m_viewportH) * 2.0f - 1.0f;
240  float zClip = 2.0f * vWin.z() - 1.0f;
241 
242  // original code from gluUnproject
243  //Transformation of normalized coordinates between -1 and 1
244  // in[0]=(winx-(float)viewport[0])/(float)viewport[2]*2.0-1.0;
245  // in[1]=(winy-(float)viewport[1])/(float)viewport[3]*2.0-1.0;
246  // in[2]=2.0*winz-1.0;
247  // in[3]=1.0;
248 
249  QVector3D result = QVector3D(xClip, yClip, zClip);
250  ShaderDebugger::debugVector3D(result, "Vector in clip space:");
251  return result;
252 }

259 void GLESRenderer::calculateMousePoints(QVector3D * nearPoint, QVector3D * farPoint, const QPoint & mousePos)
260 {
261  float winX = m_viewportX + mousePos.x();
262  float winY = m_viewportY + (m_viewportH - mousePos.y());
263  * nearPoint = unProjectViewportPoint(QVector3D(winX, winY, 0.0));
264  * farPoint = unProjectViewportPoint(QVector3D(winX, winY, 1.0));
265  // qDebug() << "nearPoint: " << *nearPoint << " farPoint: " << *farPoint;
266 }

273 float GLESRenderer::distanceToMouseClick(QVector3D p, const QPoint & mousePos)
274 {
275  QVector3D nearPoint, farPoint;
276  calculateMousePoints(&nearPoint, &farPoint, mousePos);
277 
278  QVector3D lineVector = nearPoint - farPoint;
279  float area = QVector3D::crossProduct(p - nearPoint, lineVector).length();
280  float distance = area / lineVector.length();
281 
282  return distance;
283 }

293 bool GLESRenderer::mouseIntersection(QVector3D * intersection, QVector3D normal, float d, const QPoint & mousePos)
294 {
295  float m0, m15;
296  QVector3D pNear, pFar; //mouse intersections on near and far clipping plane
297  calculateMousePoints(&pNear, &pFar, mousePos);
298  QVector3D v = pFar - pNear; //vector from near to far clipping plane
299  m15 = d + QVector3D::dotProduct(normal, pNear);
300  m0 = QVector3D::dotProduct( normal, v);
301  if(m0 == 0.0f)
302  return false;
303  float lambda = -m15 / m0;
304  if(lambda > 1.0f || lambda < 0.0f) //we have no intersection in frustum
305  return false;
306  * intersection = pNear + lambda * v;
307  return true;
308 }

508 bool GLESRenderer::initialize()
509 {
510  if(m_initialized)
511  return true;
512 #ifdef USE_QOPENGL_FUNCTIONS
513  //QOpenGLFunctions::initializeOpenGLFunctions();
514 #endif
515  //Setup shaders and program
516  m_vShader = new QOpenGLShader(QOpenGLShader::Vertex,this); //vertex shader
517  m_vShader->compileSourceFile(m_vShaderFileName);
518  if(!m_vShader->isCompiled())
519  {
520  qDebug("GLESRenderer::initialize: Compiling vertex shader failed. Log follows:\n%s",
521  qPrintable(m_vShader->log()));
522  return false;
523  }
524  m_fShader = new QOpenGLShader(QOpenGLShader::Fragment,this); // fragment shader
525  m_fShader->compileSourceFile(m_fShaderFileName);
526  if(!m_fShader->isCompiled())
527  {
528  qDebug("GLESRenderer::initialize: Compiling fragment shader failed. Log follows:\n%s",
529  qPrintable(m_fShader->log()));
530  return false;
531  }
532 
533  m_renderProgram = new QOpenGLShaderProgram(this);
534  m_renderProgram->addShader(m_vShader);
535  m_renderProgram->addShader(m_fShader);
536  m_renderProgram->link();
537  if(!m_renderProgram->isLinked())
538  {
539  qDebug("GLESRenderer::initialize: Linking program failed. Log follows:\n%s",
540  qPrintable(m_renderProgram->log()));
541  return false;
542  }
543 
544  // Get all locations of shader variables
545  //Get locations of attributes and uniforms
546  //Non existing attributes and uniforms will return -1
547  //Attributes
548  m_location_aVertex = m_renderProgram->attributeLocation("a_Vertex");
549  m_location_aColor = m_renderProgram->attributeLocation("a_Color");
550  m_location_aNormal = m_renderProgram->attributeLocation("a_Normal");
551  m_location_aTexCoord = m_renderProgram->attributeLocation("a_TexCoord");
552  //Uniforms
553  //flags
554  m_location_uLightingEnabled = m_renderProgram->uniformLocation("u_LightingEnabled");
555  m_location_uColorArrayEnabled = m_renderProgram->uniformLocation("u_ColorArrayEnabled");
556  m_location_uTextureEnabled = m_renderProgram->uniformLocation("u_TextureEnabled");
557  m_location_uMaskActive = m_renderProgram->uniformLocation("u_MaskActive");
558  //matrices
559  m_location_uNormalMatrix = m_renderProgram->uniformLocation("u_NormalMatrix");
560  m_location_uMvpMatrix = m_renderProgram->uniformLocation("u_MvpMatrix");
561  //lighting
562  m_location_uAmbientAndDiffuseColor = m_renderProgram->uniformLocation("u_AmbientAndDiffuseColor");
563  m_location_uAmbientLightBrightness = m_renderProgram->uniformLocation("u_AmbientLightBrightness");
564  m_location_uLightDirection = m_renderProgram->uniformLocation("u_LightDirection");
565  m_location_uSpecularColor = m_renderProgram->uniformLocation("u_SpecularColor");
566  m_location_uShininess = m_renderProgram->uniformLocation("u_Shininess");
567  m_location_uHalfPlaneVector = m_renderProgram->uniformLocation("u_HalfPlaneVector");
568  //texture
569  m_location_uTextureSampler = m_renderProgram->uniformLocation("s_Texture");
570  //point size
571  m_location_uPointSize = m_renderProgram->uniformLocation("u_PointSize");
572  //mask
573  m_location_uMaskDiameter = m_renderProgram->uniformLocation("u_MaskDiameterSquare");
574  m_location_uMaskCenter = m_renderProgram->uniformLocation("u_MaskCenter");
575 
576 #ifdef DEBUG_GLESRENDERER
577  ShaderDebugger::setEnabled(true);
578  ShaderDebugger::debugUniforms(m_renderProgram->programId());
579  ShaderDebugger::setEnabled(false);
580 #endif
581 
582  //get present viewport settings
583  readGLViewportSettings();
584  m_initialized = true;
585  return true;
586 }

592 bool GLESRenderer::bind()
593 {
594  bool ok = true;
595  if(!m_initialized)
596  ok = initialize();
597  if(!ok)
598  return false;
599  m_renderProgram->bind();
600  //Activate uniforms
601  //flags
602  if(m_location_uColorArrayEnabled != -1)
603  m_renderProgram->setUniformValue(m_location_uColorArrayEnabled, m_colorArrayEnabled);
604  if(m_location_uLightingEnabled != -1)
605  m_renderProgram->setUniformValue(m_location_uLightingEnabled, m_lightingEnabled);
606  if(m_location_uTextureEnabled != -1)
607  m_renderProgram->setUniformValue(m_location_uTextureEnabled, m_textureEnabled);
608  if(m_location_uMaskActive != -1)
609  m_renderProgram->setUniformValue(m_location_uMaskActive, m_maskActive);
610  //matrices
611  if( m_location_uNormalMatrix != -1)
612  m_renderProgram->setUniformValue(m_location_uNormalMatrix, m_normalMatrix);
613  if(m_location_uMvpMatrix != -1)
614  m_renderProgram->setUniformValue(m_location_uMvpMatrix, m_mvpMatrix);
615  //lighting
616  if(m_location_uAmbientAndDiffuseColor != -1)
617  m_renderProgram->setUniformValue(m_location_uAmbientAndDiffuseColor,
618  m_ambientAndDiffuseColor.red(), m_ambientAndDiffuseColor.green(),
619  m_ambientAndDiffuseColor.blue(), m_ambientAndDiffuseColor.alpha());
620  if(m_location_uAmbientLightBrightness != -1)
621  m_renderProgram->setUniformValue(m_location_uAmbientLightBrightness, m_ambientLightBrightness);
622  if(m_location_uLightDirection != -1)
623  m_renderProgram->setUniformValue(m_location_uLightDirection, m_lightDirection);
624  if(m_location_uSpecularColor != -1)
625  m_renderProgram->setUniformValue(m_location_uSpecularColor,
626  m_specularColor.red(), m_specularColor.green(),
627  m_specularColor.blue(), m_specularColor.alpha());
628  if(m_location_uShininess != -1)
629  m_renderProgram->setUniformValue(m_location_uShininess, m_shininess);
630  if(m_location_uHalfPlaneVector != -1)
631  m_renderProgram->setUniformValue(m_location_uHalfPlaneVector, m_halfPlaneVector);
632  //texture
633  if(m_location_uTextureSampler != -1)
634  m_renderProgram->setUniformValue(m_location_uTextureSampler, 0); //set sampler to use texture unit 0
635  //point size
636  if(m_location_uPointSize != -1)
637  m_renderProgram->setUniformValue(m_location_uPointSize, m_pointSize);
638 #ifndef GLES
639  //glPointSize(m_pointSize); //set point size independent of vertex shader
640 #endif
641  if(m_location_uMaskDiameter != -1)
642  m_renderProgram->setUniformValue(m_location_uMaskDiameter, m_maskDiameter * m_maskDiameter);
643  if(m_location_uMaskCenter != -1)
644  m_renderProgram->setUniformValue(m_location_uMaskCenter,QVector2D(m_viewportW / 2, m_viewportH /2));
645 
646  m_bound = true;
647  return m_bound;
648 }
